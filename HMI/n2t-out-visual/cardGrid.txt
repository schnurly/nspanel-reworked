            {
                sleepValue+=1

                {
                    sleepValue=0
                    dim=dimValue
                    tSend.txt="event,sleepReached,cardGrid"
                    //send calc crc
                    btlen tSend.txt,sys0
                    crcrest 1,0xffff // reset CRC
                    crcputh 55 bb
                    crcputs sys0,2
                    crcputs tSend.txt,0
                    //send cmd

                    prints sys0,2
                    prints tSend.txt,0
